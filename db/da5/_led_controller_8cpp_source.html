267 void LedController::spiTransfer(unsigned int segment, byte opcode, byte data) {
268  if (!initilized || segment >= conf.SegmentCount) {
269  return;
270  }
271 
272  // Create an array with the data to shift out
273  unsigned int offset = segment * 2;
274  unsigned int maxbytes = conf.SegmentCount * 2;
275 
276  for (unsigned int i = 0; i < maxbytes; i++) {
277  spidata[i] = 0x00;
278  }
279 
280  // put our device data into the array
281  spidata[offset + 1] = opcode;
282  spidata[offset] = data;
283 
284  // enable the line
285  digitalWrite(conf.SPI_CS, LOW);
286 
287  if (conf.useHardwareSpi) {
288  SPI.beginTransaction(SPISettings(8000000, MSBFIRST, SPI_MODE0));
289  }
290 
291  // Now shift out the data
292  for (int i = maxbytes; i > 0; i--) {
293  if (conf.useHardwareSpi) {
294  SPI.transfer(spidata[i - 1]);
295  } else {
296  shiftOut(conf.SPI_MOSI, conf.SPI_CLK, MSBFIRST, spidata[i - 1]);
297  }
298  }
299 
300  if (conf.useHardwareSpi) {
301  SPI.endTransaction();
302  }
303 
304  // latch the data onto the display
305  digitalWrite(conf.SPI_CS, HIGH);
306 }
